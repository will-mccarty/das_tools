#Define object and date range:
startdate=2019120100
enddate=2019123118

ps = ncd.obs_template('x0042c2.diag_conv_ps_ges.${yyyy}${mm}${dd}_${hh}z.nc4',startdate=startdate,enddate=enddate)

# set your mask for used data:
msk = '(used == 1)'
ps.set_mask(msk)
ps.use_mask(True)

# get list of unique kx values:
kxes =  np.unique(ps.v('kx'))

# loop through kxes:

for kx in kxes:
    # define and set mask
    cur_msk = '(used == 1) & (kx == {})'.format(kx)
    print('Plotting mask: '+cur_msk )
    
    ps.set_mask(cur_msk)
    
    # get time series
    cur_timeseries = ps.stat_ts('mean','omf',DataFrame=True)
    
    # plot timeseries for kx
    plt.plot(cur_timeseries['datetime'],cur_timeseries['Val'],label='{}'.format(kx))
    
# Show current plot
plt.legend()
plt.xticks(rotation=45)
plt.show
